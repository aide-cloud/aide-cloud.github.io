# 4. for循环

## 语法

```go
# 写法一
for <布尔表达式> {
    fmt.Println("A")
}

# 写法二
for i := 0; i < 10; i++ {
    fmt.Println("A")
}

# 写法三
array := [5]int{1, 2, 3, 4, 5}
for index, value := range array {
    fmt.Println(index, value)
}

# 写法四
str := "hello"
for index, value := range str {
    fmt.Println(index, value)
}

# 写法五
myMap := map[string]string{"a": "A", "b": "B"}
for index, value := range myMap {
    fmt.Println(index, value)
}

# 写法六
slice := []int{1, 2, 3, 4, 5}
for index, value := range slice {
    fmt.Println(index, value)
}

# 写法七
ch := make(chan int, 10)
for index, value := range ch {
    fmt.Println(index, value)
}
```

## 示例

### 字符串遍历
```go
package main

import "fmt"

func main() {
    str := "hello"
    for index, value := range str {
        fmt.Println(index, value)
    }
}
```

### 无限循环
```go
package main

import "fmt"

func main() {
    for {
        fmt.Println("A")
    }
}
```

### 定长循环
```go
package main

import "fmt"

func main() {
    for i := 0; i < 10; i++ {
        fmt.Println("i = ", i)
    }
}
```

### 条件循环
```go
package main

import "fmt"

func main() {
    i := 0
    for i < 10 {
        fmt.Println("i = ", i)
        i++
    }
}
```

## 关键字
```go
break
continue
goto
```

### break
```go
package main

import "fmt"

func main() {
    for i := 0; i < 10; i++ {
        if i == 5 {
            break
        }
        fmt.Println("i = ", i)
    }
    // i = 0
    // i = 1
    // i = 2
    // i = 3
    // i = 4
}
```

### continue
```go
package main

import "fmt"

func main() {
    for i := 0; i < 10; i++ {
        if i == 5 {
            continue
        }
        fmt.Println("i = ", i)
    }
    // i = 0
    // i = 1
    // i = 2
    // i = 3
    // i = 4
    // i = 6
    // i = 7
    // i = 8
    // i = 9
}
```

### goto
```go
package main

import "fmt"

func main() {
    i := 0
    LOOP:
        fmt.Println("i = ", i)
        i++
        if i < 10 {
            goto LOOP
        }
    // i = 0
    // i = 1
    // i = 2
    // i = 3
    // i = 4
    // i = 5
    // i = 6
    // i = 7
    // i = 8
    // i = 9
}
```
