                                                                                                      # 1. 基础命令

                                                                                                      # 介绍

                                                                                                      Linux 是一个类 Unix 的操作系统，它是免费使用和自由传播的，因此成为类 Unix 操作系统中最流行的一个。Linux 继承了 Unix 以网络为核心的设计思想，是一个性能稳定的多用户网络操作系统。Linux 可以运行主要的 Unix 工具软件、应用程序和网络协议。它支持 32 位和 64 位硬件。

                                                                                                      # 常用命令

                                                                                                      # 查看系统版本

                                                                                                      cat /etc/redhat-release

                                                                                                      # 查看系统内核版本

                                                                                                      uname -a

                                                                                                      # 查看系统版本

                                                                                                      cat /proc/version

                                                                                                      # 查看磁盘空间

                                                                                                      df -h

                                                                                                      # 查看内存使用情况

                                                                                                      free -m

                                                                                                      # 查看 CPU 信息

                                                                                                      cat /proc/cpuinfo

                                                                                                      # 查看 CPU 核数

                                                                                                      cat /proc/cpuinfo | grep "processor" | wc -l

                                                                                                      # 查看 CPU 使用率

                                                                                                      top

                                                                                                      # 查看系统运行时间、用户数、负载

                                                                                                      uptime

                                                                                                      # 查看系统所有用户

                                                                                                      cat /etc/passwd

                                                                                                      # 查看系统所有组

                                                                                                      cat /etc/group

                                                                                                      # 查看系统所有进程

                                                                                                      ps -ef

                                                                                                      # 查看系统所有端口

                                                                                                      netstat -tunlp

                                                                                                      # 查看系统所有服务

                                                                                                      chkconfig --list

service --status-all

systemctl list-unit-files --type=service

systemctl list-units --type=service

                                                                                                      # 查看系统所有开机启动项

                                                                                                      chkconfig --list

                                                                                                      # 查看系统所有已安装软件

                                                                                                      rpm -qa  # RedHat 系

yum list installed # RedHat 系

dpkg -l # Debian 系

apt list --installed # Debian 系

pacman -Q # Arch 系

zypper search -i # SUSE 系

                                                                                                      # 查看系统所有已安装软件包

                                                                                                      rpm -qa | grep xxx

                                                                                                      # 网络相关

                                                                                                      # 查看本机 IP 地址

                                                                                                      ifconfig

ip addr

ip a

                                                                                                      # 查看本机 MAC 地址

                                                                                                      ifconfig eth0 | grep HWaddr | awk '{print $5}' # RedHat 系

ifconfig eth0 | grep ether | awk '{print $2}' # Debian 系

ip link show eth0 | grep ether | awk '{print $2}' # Arch 系

                                                                                                      # 查看本机默认网关

                                                                                                      route -n

                                                                                                      # 查看本机 DNS 服务器

                                                                                                      cat /etc/resolv.conf

                                                                                                      # 查看本机主机名

                                                                                                      hostname

                                                                                                      # 查看本机域名

                                                                                                      domainname

                                                                                                      # 查看本机所有网卡流量

                                                                                                      ifconfig -a

                                                                                                      # 查看本机指定网卡流量

                                                                                                      ifconfig eth0

                                                                                                      # 文件相关

                                                                                                      # 查看文件内容

                                                                                                      cat /etc/passwd

                                                                                                      # 查看文件内容（分页）

                                                                                                      more /etc/passwd

                                                                                                      # 查看文件内容（分页，可向前翻页）

                                                                                                      less /etc/passwd

                                                                                                      # 查看文件内容（十六进制）

                                                                                                      od -x /etc/passwd

                                                                                                      # 查看文件内容（二进制）

                                                                                                      od -b /etc/passwd

                                                                                                      # 查看文件内容（八进制）

                                                                                                      od -o /etc/passwd

                                                                                                      # 查看文件内容（十进制）

                                                                                                      od -d /etc/passwd

                                                                                                      # 进程管理

                                                                                                      # 查看进程

                                                                                                      ps -ef

                                                                                                      # 查看进程（树状）

                                                                                                      pstree

                                                                                                      # 查看进程（树状，显示 PID）

                                                                                                      pstree -p

                                                                                                      # 查看进程（树状，显示进程名）

                                                                                                      pstree -a

                                                                                                      # 查杀进程

                                                                                                      kill -9 PID

ps -ef |grep xxx
kill -9 PID

# 万能查杀
kill -9 $(ps -ef|grep 'keyword'|grep -v grep|awk '{print $2}')
# or
ps -ef |grep 'keyword' |awk '{print $2}'|xargs kill -9

                                                                                                      # 查看进程打开的文件

                                                                                                      lsof -p PID

                                                                                                      # 查看进程打开的端口

                                                                                                      netstat -tunlp | grep PID

                                                                                                      # 查看进程打开的端口（树状）

                                                                                                      lsof -i -P -n | grep PID
